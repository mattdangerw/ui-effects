Clutter.InputDevice

Generic representation of an input device. The actual contents of this structure depend on the backend used.

from gi.repository import Clutter

input_device = Clutter.InputDevice(backend=value, device_manager=value, device_mode=value, device_type=value, enabled=value, has_cursor=value, id=value, name=value)
